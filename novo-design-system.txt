<script>
    // Particle background logic
    const canvas = document.createElement('canvas');
    canvas.style.position = 'fixed';
    canvas.style.top = '0';
    canvas.style.left = '0';
    canvas.style.width = '100%';
    canvas.style.height = '100%';
    canvas.style.pointerEvents = 'none';
    canvas.style.zIndex = '0';
    canvas.style.opacity = '0.2';
    document.body.appendChild(canvas);

    const ctx = canvas.getContext('2d');
    let width, height;

    function resize() {
        width = canvas.width = window.innerWidth;
        height = canvas.height = window.innerHeight;
    }

    window.addEventListener('resize', resize);
    resize();

    const dots = [];
    for(let i=0; i<40; i++) {
        dots.push({
            x: Math.random() * width,
            y: Math.random() * height,
            r: Math.random() * 1.2,
            vx: (Math.random() - 0.5) * 0.15,
            vy: (Math.random() - 0.5) * 0.15
        });
    }

    function animate() {
        ctx.clearRect(0, 0, width, height);
        ctx.fillStyle = '#4be277';
        dots.forEach(dot => {
            dot.x += dot.vx;
            dot.y += dot.vy;
            if(dot.x < 0 || dot.x > width) dot.vx *= -1;
            if(dot.y < 0 || dot.y > height) dot.vy *= -1;
            ctx.beginPath();
            ctx.arc(dot.x, dot.y, dot.r, 0, Math.PI * 2);
            ctx.fill();
        });
        requestAnimationFrame(animate);
    }
    animate();

    // Table row interaction simulation
    document.querySelectorAll('tbody tr').forEach(row => {
        row.addEventListener('click', (e) => {
            if (e.target.closest('button')) return;
            document.querySelectorAll('tbody tr').forEach(r => r.classList.remove('table-row-selected'));
            row.classList.add('table-row-selected');
            const name = row.querySelector('.font-bold').innerText;
            const id = row.querySelector('.font-mono-sm').innerText;
            const editTitle = document.querySelector('#edit-modal .font-label-md.font-bold.text-primary');
            if (editTitle) editTitle.innerText = `EDITANDO: ${name}`;
            const editId = document.querySelector('#edit-modal .text-on-surface-variant.opacity-50.font-mono-sm');
            if (editId) editId.innerText = id;
        });
    });
</script>
